3.1SOLID Principles

3.1.1 S - Single Responsibility Principle

Foundation
// BAD: OrderService handles orders, emails, and PDFs
public class OrderService {
    public Order createOrder(OrderRequest req) { ... }
    public void sendConfirmationEmail(Order order) { ... }  // Not its job
    public byte[] generateInvoicePdf(Order order) { ... }   // Not its job
}

// GOOD: Each class has one reason to change
public class OrderService {
    private final OrderRepository repo;
    private final EventPublisher events;

    public Order createOrder(OrderRequest req) {
        Order order = Order.from(req);
        repo.save(order);
        events.publish(new OrderCreatedEvent(order)); // Others react to event
        return order;
    }
}
// EmailService listens to OrderCreatedEvent -> sends email
// InvoiceService listens to OrderCreatedEvent -> generates PDF

3.1.2 O - Open/Closed Principle

Foundation
// Open for extension, closed for modification
// BAD: Adding new payment type requires modifying existing code
public BigDecimal calculateFee(Payment payment) {
    if (payment.type == CREDIT_CARD) return amount.multiply(0.029);
    else if (payment.type == PAYPAL) return amount.multiply(0.034);
    // Must modify this method for every new payment type
}

// GOOD: Strategy pattern - extend without modifying
public interface FeeCalculator {
    BigDecimal calculate(BigDecimal amount);
    PaymentType supports();
}

@Component
public class CreditCardFeeCalculator implements FeeCalculator {
    public BigDecimal calculate(BigDecimal amount) { return amount.multiply(new BigDecimal("0.029")); }
    public PaymentType supports() { return PaymentType.CREDIT_CARD; }
}
// Add new payment type = add new class, zero changes to existing code

3.1.3 L - Liskov Substitution Principle

Foundation
// Subtypes must be substitutable for their base types without breaking behavior
// Classic violation: Square extends Rectangle
// BAD:
class Rectangle { void setWidth(int w); void setHeight(int h); }
class Square extends Rectangle {
    void setWidth(int w) { super.setWidth(w); super.setHeight(w); } // Breaks Rectangle contract
}

// GOOD: Use composition or separate hierarchy
sealed interface Shape permits Rectangle, Square, Circle {
    double area();
}
record Rectangle(double width, double height) implements Shape {
    public double area() { return width * height; }
}
record Square(double side) implements Shape {
    public double area() { return side * side; }
}

3.1.4 I - Interface Segregation Principle

Foundation
// BAD: Fat interface forces unnecessary implementations
interface UserRepository {
    User findById(String id);
    List<User> findAll();
    void save(User user);
    void delete(String id);
    List<User> findByComplexCriteria(Criteria c); // Not all clients need this
    void bulkImport(List<User> users);             // Only admin needs this
}

// GOOD: Segregated interfaces
interface UserReader { User findById(String id); List<User> findAll(); }
interface UserWriter { void save(User user); void delete(String id); }
interface UserBulkOps { void bulkImport(List<User> users); }
// Clients depend only on what they use

3.1.5 D - Dependency Inversion Principle

Foundation
// High-level modules should not depend on low-level modules.
// Both should depend on abstractions.

// BAD: OrderService directly depends on MySQLOrderRepository
public class OrderService {
    private MySQLOrderRepository repo = new MySQLOrderRepository(); // Tight coupling
}

// GOOD: Depend on abstraction
public class OrderService {
    private final OrderRepository repo; // Interface - can be MySQL, Mongo, InMemory

    public OrderService(OrderRepository repo) { this.repo = repo; }
}
// In tests: new OrderService(new InMemoryOrderRepository())
// In prod: Spring injects MySQLOrderRepository

Interview Tip

When asked about SOLID, always give the principle + a real violation you've seen in production + how you refactored it. Shows practical experience, not just textbook knowledge.
